11 // Features:
12 // - Wraps inotify watch instances in a C++ class.
13 // - Can set up inotify watches on existing directories (not files).
14 // - Can handle recursive subdirectory watches (with pathnames not ending with '/')
15 // automatically.
16 // - Can handle moving watches dynamically and efficiently; watches are removed from all
17 // moving-out directories and added to all moving-in directories.
18 // - If a whole directory tree is copied into a watch at once, it will not miss reporting
19 // any member of the tree, which was an intrinsic problem of inotify system calls.
20 // - Utf-8 (such as hangul) pathnames handle well.
21 // - All run-time errors including system call errors are logged and thrown as
22 // std::system_error exception.
23 // - Can accept user-provided logging functions, like fprintf(stderr, ...) and syslog().
24 // - Supports timed waits in reading inotify events.

96 template <typename Log>
97  // The parameter Log is type of a function (object), void (*)(const char*...), that
98  // is used for logging.
99 class Inotify {
100  const Log& log; // a function (object) for logging
101 
102  const int fd; // inotify file descriptor associated with this inotify instance

147 template <typename Log>
148 int Inotify<Log>::add_watch(const std::string& path, bool in_move)
149 // The given path is required to be non-empty string for an existing directory.
150 // Otherwise, it will be ignored, but with an error logged.
151 // If the path already contains child files and subdirectories in it, the in_move flag
152 // comes in play. If true and if the path is recursive (that is, not ending with '/'),
153 // every subdirectory (but not file) will have its own watch set up recursively and
154 // implicitly without notifying. If false, regardless of whether the path is recursive or
155 // not, every (immediate) child file and subdirectory will be prepared to be reported as
156 // IN_CREATE by the next call to read() (but, in this case, read() may duplicately report
157 // the same IN_CREATE event for some files or directories, depending on how fast the path
158 // watch kicks in).
159 // Note that the in_move flag is not just for IN_MOVED_TO'd watches, but also can be used
160 // for initial setup of watches on existing directories, which is why its default
161 // argument is set true.
162 // Todo: watch for non-existing directory/file yet.
163 // Todo: negative watch specification.
164 {
165  const bool recursive = path.back() != '/';
166  // will be always true if called from read().
167  const int wd = inotify_add_watch(fd, path.c_str(),
168  mask | IN_ONLYDIR | IN_MOVE_SELF | (recursive ? IN_CREATE | IN_MOVED_TO : 0));
169  // IN_ONLYDIR is to set up a watch on directory only.
170 
171  if ( wd == -1 ) { // if non-directory, non-existing, or without read-permission,
172  //log("Warning: Cannot watch \"%s\": %m", path.c_str());
173  // We can use "%m" for strerror(errno) if log is Syslog<> function object.
174  log("Warning: Cannot watch \"%s\": %s", path.c_str(), std::strerror(errno));
175  return wd;
176  }
177 
178  const auto it = watches.find(wd);
179  if ( it == watches.end() ) {
180  printf("[%d] %s created\n", wd, path.c_str());
181  watches.emplace(wd, Watch { path, false });
182  }
183 
184  else { // if the watch was already registered,
185  // The above inotify_add_watch() will return wd of an existing watch if there
186  // already exists a watch for the given path, in which case we determine the
187  // watch is moved rather than created newly.
188 
189  const std::string& path0 = it->second.path;
190  const auto& diff = std::mismatch(path0.begin(), path0.end(), path.begin());
191  // finds the first mismatched characters in comparing path0 and path.
192  if ( diff.first == path0.end() &&
193  (diff.second == path.end() || !recursive && diff.second+1 == path.end()) ) {
194  // Do nothing if path == path0 or path == path0 + "/".
195  printf("[%d] %s ignored as a duplicate\n", wd, path.c_str());
196  return wd;
197  }
198  printf("[%d] %s %s\n", wd, path.c_str(),
199  //in_move ? "moved" : "changed to recursive");
200  // commented-out because in_move can be true for initial setup of watches
201  // on existing directories.
202  diff.second == path.end() && diff.first+1 == path0.end() &&
203  *diff.first == '/' ? "changed to recursive" : "moved");
204  it->second.path = path;
205  //it->second.in_move = false; // not necessary
206  }
207 
208  // Note, when a directory that is either already a watch or not is moved into another
209  // watch, its children come into existence at the same time of its move and we will
210  // never get a report from it for the existence of any its children although its
211  // watch (if it is a watch) kicks in immediately on its move, and even if it has any
212  // subdirectories that are also a watch we will not get any report for its
213  // grand-children since they are also moved at the same time as it is moved. If it is
214  // not a watch, we can only set up its watch, no matter how fast we do, after all its
215  // children have been moved. So, we will get notified of only the top directory in
216  // this case.
217 
218  // When a non-empty directory is copied into a watch together with its children,
219  // however, we may get notified of the existence of any its children, depending on
220  // how fast we set up its watch. If some children are already copied before we set up
221  // the watch, we will miss reports for those children. If it has any grand-children,
222  // we will get their reports only after we set up a watch for their corresponding
223  // parent and only for such grand-children that are copied after their parent watch
224  // kicks in. This is because the inotify watches are basically non-recursive and
225  // effective only for their immediate children. So, in this case, we may get reports
226  // for some children and may miss for other children.
227 
228  // In case we are IN_MOVED_To'd, we traverse all the subdirectories (but not files)
229  // and set up their watches recursively and implicitly, without reporting to the
230  // read().
231  if ( in_move ) { // if we are IN_MOVED_TO'd,
232  if ( recursive )
233  // We create a watch for every subdirectory down below, but without reporting
234  // it to read().
235  for ( const fs::path& subdir: fs::directory_iterator(path) )
236  // The directory_iterator() here will always succeed, since the previous
237  // check for "wd == -1" filters out what is not this case.
238  if ( fs::is_directory(subdir) /* && !fs::is_symlink(subdir) */ )
239  // The is_directory() and the is_symlink() are mutually exclusive.
240  add_watch(subdir.string(), true);
241  }
242 
243  // If we are IN_CREATEd, we traverse our immediate children (both files and
244  // subdirectories) and we prepare them to be reported as IN_CREATEd by the read(),
245  // which may possibly get a duplicate event for some of them from the system but will
246  // not miss any. We do not recurse into any child subdirectories here because read()
247  // will do, but a duplicate recurse of read() due to a duplicate event from the
248  // system will be checked out by the previous "ignored as a duplicate" filtering.
249  else { // if we are IN_CREATEd,
250  for ( const fs::path& path: fs::directory_iterator(path) )
251  if ( !fs::is_other(path) ) {
252  // if a directory, a regular file, or a symlink, but not device file,
253  // fifo, or socket,
254 
255  // Make a new IN_CREATE event to be read().
256  const bool isdir = fs::is_directory(path);
257  if ( mask & IN_CREATE || isdir && recursive ) {
258  const std::string& name = path.filename().string();
259  const uint32_t len =
260  (name.size()+sizeof(int))/sizeof(int)*sizeof(int);
261  // length including '\0' that fits in word boundary
262  inotify_event& event =
263  *(inotify_event*)((char*)buffer + bytes_in_buffer);
264  bytes_in_buffer += sizeof(inotify_event) + len;
265  if ( bytes_in_buffer > sizeof(buffer) ) {
266  log("Error: add_watch() - Buffer underruns");
267  throw std::system_error(EINVAL, std::system_category());
268  }
269  event.wd = wd;
270  event.mask = isdir ? IN_ISDIR | IN_CREATE : IN_CREATE;
271  event.cookie = 0;
272  event.len = len;
273  name.copy(event.name, len);
274  event.name[name.size()] = '\0';
275  }
276  }
277  }
278 
279  return wd; // return wd of only the top directory
280 }

300 template <typename Log>
301 const inotify_event* Inotify<Log>::read(int timeout, int read_delay)
302 // Read one inotify event from fd, or return nullptr if timed out.
303 // Will throw an exception when an error is returned from poll() or read().
304 
305 // The read_delay: The time in milliseconds to wait after the first event arrives before
306 // reading the buffer. This allows further events to accumulate before reading, which
307 // allows the kernel to consolidate like events and can enhance performance when there
308 // are many similar events. The read_delay should be in [0..1000], or may cause a
309 // run-time exception.
310 // If no watches are set up, read() will still run ok and will wait for nothing.
311 // Todo: Handle IN_Q_OVERFLOW and restart the daemon.
312 {
313  const auto then = std::chrono::system_clock::now(); // check starting time
314 
315  // If the buffer underruns, (re)fill it by reading more events from read().
316  if ( bytes_in_buffer == 0 ) {
317  const char* where;
318 
319  where = "poll()";
320  switch ( poll(&fds, 1, timeout) ) {
321  case 0: // timed out!
322  return nullptr;
323 
324  default: // or, "case 1:" and events are ready!
325  where = "usleep()";
326  if ( usleep(read_delay*1000u) != -1 ) {
327  where = "read()";
328  bytes_in_buffer = ::read(fd, buffer, sizeof(buffer));
329  if ( bytes_in_buffer > 0 )
330  break;
331  if ( bytes_in_buffer == 0 )
332  // EOF reached. Possibly too many events occurred at once?
333  errno = EIO;
334  }
335  // intentional fall-through
336 
337  case -1: // error in system call
338  log("Error: %s:%d - %s", where, errno, std::strerror(errno));
339  throw std::system_error(errno, std::system_category());
340  }
341  }
342 
343  do {
344  const inotify_event& event =
345  *(inotify_event*)((char*)buffer + bytes_handled);
346 
347  bytes_handled += sizeof(inotify_event) + event.len;
348  if ( bytes_handled >= bytes_in_buffer ) {
349  if ( bytes_handled > bytes_in_buffer ) { // sanity check
350  // An incomplete event was read from read(). This is not to happen, but
351  // we just report the same failure that read() would do in case the
352  // buffer is too small to hold an event.
353  log("Error: read() - Incomplete event returned");
354  throw std::system_error(EINVAL, std::system_category());
355  }
356  bytes_handled = bytes_in_buffer = 0;
357  }
358 
359  const auto it = watches.find(event.wd);
360  if ( it == watches.end() ) { // sanity check
361  log("Error: read() - Event for unknown wd [%d] possibly due to IN_Q_OVERFLOW",
362  event.wd);
363  throw std::system_error(EINVAL, std::system_category());
364  }
365  Watch& watch = it->second;
366  printf("- [%d] %s (%#x)\n", event.wd,
367  (event.len ? watch.path/event.name : watch.path).c_str(), event.mask);
368 
369  // A new subdirectory was created or moved in.
370  if ( event.mask & (IN_CREATE | IN_MOVED_TO) &&
371  event.mask & IN_ISDIR &&
372  watch.path.back() != '/' ) {
373  const bool in_move = event.mask & IN_MOVED_TO; // will cast to 0 or 1.
374  const int wd = add_watch(watch.path/event.name, in_move);
375  // The event.name here will be non-empty for IN_CREATE and IN_MOVED_TO.
376  // When a watch is moved into another directory, the watch is retained only
377  // if that directory is also a watch and recursive, or deleted otherwise (at
378  // the next IN_MOVE_SELF event). And, even if the watch was not recursive, it
379  // turns into a recursive one and all its subdirectories as well. This way,
380  // every subdirectory of a recursive watch directory is always a recursive
381  // watch.
382 
383  // Actually, we should think of the IN_MOVED_TO event as two separate events;
384  // deleting the old watch, and creating a new watch because put under a
385  // recursive watch directory. However, the inotify simply reuses (recycles)
386  // the same watch in this case for efficiency reasons, and we just mark this
387  // recycle on the in-struct in_move flag. If a watch is marked with this
388  // recycle, it will survive the next IN_MOVE_SELF and will not be deleted.
389  if ( in_move && wd >= 0 /* != -1 */ )
390  watches.at(wd).in_move = true;
391  // In case of in_move, we here mark only the top directory of the moved
392  // tree as in_move, because the IN_MOVED_TO event and the corresponding
393  // IN_MOVE_SELF event (if any) arrive only with the top directory, but
394  // not recursively.
395  }
396 
397  // Delete a watch recursively unless in move.
398  if ( event.mask & IN_MOVE_SELF ) {
399  // We do not need to check here if it is a directory because only directory
400  // watches were allowed.
401  if ( watch.in_move )
402  // Do not delete if marked as in_move.
403  watch.in_move = false;
404 
405  else if ( watch.path.back() == '/' )
406  rm_watch(event.wd);
407 
408  else
409  // We recursively delete this MOVE_SELF'd watch and all the watches for
410  // its subdirectories (whether or not they are recursive watches).
411  for ( const auto& it: watches )
412  if ( it.second.path.compare(0, watch.path.size(), watch.path) == 0 )
413  // checks if it.second.path starts with watch_path. Even if
414  // it.second.path is a prefix of watch.path on the other hand, it
415  // will be handled ok and compare() will yield non-zero.
416  rm_watch(it.first);
417  // We do not actually remove members from the watches map here, which
418  // will be done at the IN_IGNORED event later. That is why we could
419  // declare watch.path to be a string reference rather than a copy of
420  // string.
421  }
422 
423  // A watch was deleted implicitly, so delete it from the dictionary too.
424  if ( event.mask & IN_IGNORED ) {
425  printf("[%d] %s deleted\n", event.wd, watch.path.c_str());
426  watches.erase(event.wd);
427  }
428 
429  // If a matching event is found, return it.
430  if ( event.mask & mask )
431  return &event;
432 
433  // Repeat until all bytes in the buffer are handled.
434  } while ( bytes_handled > 0 );
435 
436  // So far, We got some events and handled all of them, but we could not report any
437  // events of interest to the caller. We will take another oppertunity to read more
438  // events only if we have enough time left though, or report nothing otherwise.
439 
440  const int time_elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
441  std::chrono::system_clock::now() - then).count();
442  if ( timeout < 0 /* == -1 */ || (timeout -= time_elapsed) > 0 )
443  return read(timeout, read_delay); // The tail-call optimization will work here.
444 
445  return nullptr; // timed out!
446 }
447 }
